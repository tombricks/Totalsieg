### MOON WAR
## What are the arrays?
# MOON_WAR_TERRAIN_MAP - Number - Contains the terrain information. Should be the default array if you need to loop through the map in general.
# MOON_WAR_CONTROLLER_MAP - Country Tag / Number - Contains the owner. Number should be used in special cases only like no owner being 0.

on_actions = {
	on_startup = {
		effect = {
			resize_array = {
				array = MOON_WAR_TERRAIN_MAP
				value = 1
				size = 64
			}
			resize_array = {
				array = MOON_WAR_CONTROLLER_MAP
				value = 0
				size = 64
			}
			set_variable = { MOON_WAR_CONTROLLER_MAP^21 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^13 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^5 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^12 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^4 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^20 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^27 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^11 = GER }
			set_variable = { MOON_WAR_CONTROLLER_MAP^3 = GER }
		}
	}
}
