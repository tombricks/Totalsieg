# Death of Hitler - GER
country_event = {
	id = GER_Death_of_Hitler.1
	title = GER_Death_of_Hitler.1.T
	desc = GER_Death_of_Hitler.1.D
	picture = GFX_report_event_german_speech
	fire_only_once = yes
	
	trigger = {
		tag = GER
		has_country_leader = {
			name = "Adolf Hitler"
			ruling_only = yes
		}
		date > 1968.5.3			
	}
	
 	immediate = {
		hidden_effect = {
			set_global_flag = GER_Death_of_Hitler
			kill_ideology_leader = fascism
			kill_ideology_leader = national_socialism
			create_country_leader = {
				name = "Großdeutscher Reichstag"
				desc = "POLITICS_REICHSTAG_DESC"
				picture = "Reichstag.tga"
				expire = "1999.1.1"
				ideology = ns_party_NSDAP
				traits = {
					TITLE_Legislative_Body
				}
			}
			create_country_leader = {
				name = "Großdeutscher Reichstag"
				desc = "POLITICS_REICHSTAG_DESC"
				picture = "Reichstag.tga"
				expire = "1999.1.1"
				ideology = f_party_NSDAP
				traits = {
					TITLE_Legislative_Body
				}
			}
			news_event = {
				id = GER_Death_of_Hitler.2
				days = 1
			}
		}
	}

	option = {
		name = GER_Death_of_Hitler.1.A
		country_event = {
			id = GER_Death_of_Hitler.3
			days = 3
		}
	}

	option = {
		name = GER_Death_of_Hitler.1.B
		trigger = {
			OR = {
				is_debug = yes
				check_variable = { GER_Death_of_Hitler_SS_Influence > 49 }
			}
		}
		country_event = {
			id = GER_Death_of_Hitler.4
			days = 3
		}
	}
}
